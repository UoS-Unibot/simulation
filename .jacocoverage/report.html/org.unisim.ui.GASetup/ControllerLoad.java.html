    private void fcControllerPropertyChange(java.beans.PropertyChangeEvent evt) {//GEN-FIRST:event_fcControllerPropertyChange
        try {
            JSONCTRNNLayout jlayout = JSONCTRNNLayout.fromFile(fcController.getSelectedFile());
            lblAuthor2.setText(jlayout.getMetadata().getAuthor());
            lblTitle2.setText(jlayout.getMetadata().getTitle());
            lblDesc2.setText(jlayout.getMetadata().getDescription());
            layout=jlayout.toCTRNNLayout();
        } catch (Exception ex) {
            lblAuthor2.setText("<File not valid JSON CTRNN Layout file>");
        }
    }//GEN-LAST:event_fcControllerPropertyChange
